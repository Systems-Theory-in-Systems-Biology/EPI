Temperature ModelÔÉÅ

The temperature model is contained in eulerpi/examples/temperature/temperature.py. The model \(y(q)=60 \cos(q)-30=s(q)\) describes the temperature for a place on the earth \(y\) by using the latitude coordinates \(q\). The jacobian \({\frac{dy}{dq}}(q_i)=-60 \sin(q_i)\) can be calculated analytically.

class Temperature(Model):
    """ """

    param_dim = 1
    data_dim = 1

    PARAM_LIMITS = np.array([[0, np.pi / 2]])
    CENTRAL_PARAM = np.array([np.pi / 4.0])

    def __init__(
        self,
        central_param: np.ndarray = CENTRAL_PARAM,
        param_limits: np.ndarray = PARAM_LIMITS,
        name: Optional[str] = None,
        **kwargs,
    ) -> None:
        super().__init__(central_param, param_limits, name=name, **kwargs)

    def forward(self, param):
        low_T = -30.0
        high_T = 30.0
        res = jnp.array(
            [low_T + (high_T - low_T) * jnp.cos(jnp.abs(param[0]))]
        )
        return res

    def jacobian(self, param):
        return jnp.array([60.0 * jnp.sin(jnp.abs(param[0]))])

SpecialitiesÔÉÅ

  • Additional fixed parameters: The model includes fixed parameters self.low_T=30.0 and self.high_T=30.0. These fixed parameters are passed to the forward function separately. You can create models with different parameters by creating several model objects.

    The best way to seperate the outputs for the parametrized models is to pass a string based on the fixed_params to the attribute run_name of the inference() function.

class TemperatureWithFixedParams(Temperature):
    def __init__(
        self,
        low_T: np.double = -30.0,
        high_T: np.double = 30.0,
        name: Optional[str] = None,
        **kwargs,
    ) -> None:
        super().__init__(name=name, **kwargs)
        self.low_T = low_T
        self.high_T = high_T

    def forward(self, param):
        return self.calc_forward(param, self.high_T, self.low_T)

    def calc_forward(self, param, high_T, low_T):
        res = jnp.array(
            [low_T + (high_T - low_T) * jnp.cos(jnp.abs(param[0]))]
        )
        return res

    def jacobian(self, param):
        return self.calc_jacobian(param, self.high_T, self.low_T)

    def calc_jacobian(self, param, high_T, low_T):
        return jnp.array([(high_T - low_T) * jnp.sin(jnp.abs(param[0]))])

def test_fixed_params(temperatures):
    """ """
    model: Model = TemperatureWithFixedParams(*temperatures)
    data = importlib.resources.path(
        "eulerpi.examples.temperature", "TemperatureData.csv"
    )

    run_name = temp_name(*temperatures)

    inference(
        model,
        data,
        num_steps=100,
        run_name=run_name,
    )

Note

The functions calc_forward() is not strictly necessary. However it can help to make it work with jax.
